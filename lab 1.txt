#include <stdio.h>
#include <stdlib.h> 
#include <pthread.h>
#define NTHREADS 2 

void * quadrado (void * arg, int vector[]) {
    int ident = * (int *) arg;
    int i = (10000/(ident)) - 1;
    int j = 0;
    printf("%d", ident);
    printf("\n");
    while(j != 5000){
        vector[i] = (vector[i]*vector[i]);
        j++;
        i--;
    }
    pthread_exit(NULL);
}
int main() {
    int vector[10000];
    int c;
    for(c=0; c<10000;c++){
        vector[c] = c;
    }
    pthread_t tid[NTHREADS]; 
    int ident[NTHREADS];
     for(int i=1; i<=NTHREADS; i++) {
       ident[i-1] = i;
       if (pthread_create(&tid[i-1], NULL, quadrado, (void *)&ident[i-1], *vector)) 
          printf("ERRO -- pthread_create\n");
    }
    pthread_exit(NULL);
    return 0;
}
